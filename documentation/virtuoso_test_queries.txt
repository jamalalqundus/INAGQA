##########################################################################################
Select Query:
##########################################################################################

PREFIX dbo: <http://dbpedia.org/ontology/>       
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>    
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX foaf: <http://xmlns.com/foaf/spec/>
PREFIX xsd: <http://www.w3.org/TR/swbp-xsch-datatypes/>
PREFIX db-datatypes: <http://dbpedia.org/datatype/>       
			
			SELECT DISTINCT
			?iri as ?entity
			(str(?lbl) as ?title)
			(str(?description) as ?description)
			?assets
			?equity
			?revenue
			?netIncome
			MAX(?logo) as ?thumbnail
			(GROUP_CONCAT(distinct ?ceo_entity; separator=", ") as ?keyPerson)
			(GROUP_CONCAT(distinct ?foundLoc;separator=", ") as ?foundationPlace)
			(GROUP_CONCAT(distinct ?founder; separator=", ") as ?foundedBy)
			(GROUP_CONCAT(distinct ?product; separator=", ") as ?product)
			(GROUP_CONCAT(distinct ?industry; separator=", ") as ?industry)
			(GROUP_CONCAT(distinct ?location; separator=", ") as ?locationCity)
			(COUNT(?property) as ?propertyCount)
			WHERE {
			  ?iri a dbo:Company ;
			       dbo:abstract ?description ;
			       rdfs:label ?lbl .
			  ?lbl bif:contains "adidas"@en  .
			  FILTER( langMatches(lang(?description),"de") )
			  FILTER( langMatches(lang(?lbl),"de") )

			  OPTIONAL {?iri dbo:assets ?assets}
			  OPTIONAL {?iri dbo:equity ?equity}
			  OPTIONAL {?iri dbo:revenue ?revenue}
			  OPTIONAL {?iri dbo:netIncome ?netIncome}
			  OPTIONAL {?iri foaf:depiction|dbo:thumbnail ?logo}
			  OPTIONAL {?iri dbo:foundationPlace/foaf:name ?foundLoc}
			  OPTIONAL {?iri dbo:foundedBy/foaf:name ?founder}
			  OPTIONAL {?iri dbo:keyPerson/foaf:name ?ceo_entity}
			  OPTIONAL {?iri dbo:industry/rdfs:label ?industry
			  filter langMatches(lang(?industry),"de")}
			  OPTIONAL {?iri dbo:product/rdfs:label ?product
			  filter langMatches(lang(?product),"de")}
			  OPTIONAL {?iri dbo:locationCity/rdfs:label ?location
			  filter langMatches(lang(?location),"de")}
			  OPTIONAL {
			    VALUES ?p { dbo:assets dbo:equity foaf:depiction dbo:thumbnail dbo:foundationPlace dbo:keyPerson dbo:revenue dbo:netIncome dbo:industry dbo:product dbo:location}
			    ?iri ?p ?property .
			  }
			 
			}
			ORDER BY DESC (?propertyCount)
